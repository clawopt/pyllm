# 排名函数详解


## rank() 基础

```python
import pandas as pd
import numpy as np

df = pd.DataFrame({
    'model': ['GPT-4o', 'Claude', 'Llama', 'Qwen', 'DeepSeek', 'Gemini'],
    'MMLU': [88.7, 89.2, 84.5, 83.5, 85.3, 86.8],
})

df['rank_default'] = df['MMLU'].rank(ascending=False)
print(df[['model', 'MMLU', 'rank_default']])
```

## 六种排名方法详解

```python
import pandas as pd
import numpy as np

scores = pd.Series([95, 90, 90, 85, 85, 85, 70], name='score')

result = pd.DataFrame({'原始分数': scores})

result['average'] = scores.rank(method='average', ascending=False)

result['min'] = scores.rank(method='min', ascending=False)

result['max'] = scores.rank(method='max', ascending=False)

result['first'] = scores.rank(method='first', ascending=False)

result['dense'] = scores.rank(method='dense', ascending=False)

print(result)
```

### 各方法可视化对比

| 原始值 | average | min | max | first | dense |
|-------|---------|-----|-----|-------|-------|
| 95 | 1 | 1 | 1 | 1 | 1 |
| 90 | **2.5** | **2** | **3** | **2** | **2** |
| 90 | **2.5** | **2** | **3** | **3** | **2** |
| 85 | **5.0** | **4** | **6** | **4** | **3** |
| 85 | **5.0** | **4** | **6** | **5** | **3** |
| 85 | **5.0** | **4** | **6** | **6** | **3** |
| 70 | 7 | 7 | 7 | 7 | 4 |

**选择指南**：
- `average`：竞赛排行榜（默认，最常用）
- `min`：保守排名（"至少排第几名"）
- `max`：激进排名（"最多排到第几名"）
- `first`：需要区分同分先后时
- `dense`：显示用（排名紧凑，如奖牌：金、银、铜）

## pct=True：百分比排名

```python
import pandas as pd
import numpy as np

np.random.seed(42)

df = pd.DataFrame({
    'model': ['GPT-4o', 'Claude', 'Llama', 'Qwen', 'DeepSeek', 'Gemini',
              'Mistral', 'Phi-3', 'Yi', 'Gemma'],
    'MMLU': [88.7, 89.2, 84.5, 83.5, 85.3, 86.8, 81.2, 79.4, 80.1, 77.8],
})

df['pct_rank'] = df['MMLU'].rank(pct=True, ascending=False)

df['percentile_tier'] = pd.cut(
    df['pct_rank'],
    bins=[0, 0.1, 0.25, 0.5, 0.75, 0.9, 1.01],
    labels=['Top 10%', 'Top 25%', 'Top 50%', 'Top 75%', 'Top 90%', '其他']
)

print(df.sort_values('MMLU', ascending=False)[['model', 'MMLU', 'pct_rank', 'percentile_tier']])
```

## LLM 场景：多维度综合排名

```python
import pandas as pd
import numpy as np

np.random.seed(42)

models = ['GPT-4o', 'Claude-3.5-Sonnet', 'Llama-3.1-70B', 'Qwen2.5-72B',
          'DeepSeek-V3', 'Gemini-1.5-Pro', 'Mistral-Large', 'Phi-3-medium']

eval_data = []
for m in models:
    base = {'GPT-4o': 88, 'Claude-3.5-Sonnet': 89, 'Llama-3.1-70B': 84,
            'Qwen2.5-72B': 83, 'DeepSeek-V3': 87, 'Gemini-1.5-Pro': 86,
            'Mistral-Large': 82, 'Phi-3-medium': 78}.get(m, 80)
    eval_data.append({
        'model': m,
        'coding': round(base + np.random.randn()*3 + 5, 1),
        'reasoning': round(base + np.random.randn()*3 - 2, 1),
        'math': round(base + np.random.randn()*4 - 8, 1),
        'chinese': round(base + np.random.randn()*2 + 3, 1),
        'price_per_1M': round(np.random.uniform(0.14, 15.0), 2),
    })

eval_df = pd.DataFrame(eval_data)

for col in ['coding', 'reasoning', 'math', 'chinese']:
    eval_df[f'{col}_rank'] = eval_df[col].rank(ascending=False, method='dense').astype(int)

eval_df['price_rank'] = eval_df['price_per_1M'].rank(ascending=True, method='dense').astype(int)

eval_df['composite_score'] = (
    eval_df['coding_rank'] * 0.30 +
    eval_df['reasoning_rank'] * 0.25 +
    eval_df['math_rank'] * 0.15 +
    eval_df['chinese_rank'] * 0.15 +
    eval_df['price_rank'] * 0.15
).round(2)

eval_df['overall_rank'] = eval_df['composite_score'].rank(method='dense').astype(int)

display_cols = ['model', 'coding', 'reasoning', 'math', 'chinese',
                'composite_score', 'overall_rank']
print("=== LLM 多维度综合排名 ===")
print(eval_df[display_cols].sort_values('overall_rank').to_string(index=False))
```

### 按场景推荐最佳模型

```python
def recommend_best(df, scenario_weights):
    """根据权重推荐模型"""
    weighted = pd.Series(0.0, index=df.index)
    for col, w in scenario_weights.items():
        rank_col = f'{col}_rank' if f'{col}_rank' in df.columns else col
        if rank_col in df.columns:
            weighted += df[rank_col] * w

    best_idx = weighted.idxmin()
    return df.loc[best_idx, 'model'], weighted.min()

scenarios = {
    '代码开发': {'coding': 0.40, 'reasoning': 0.35, 'math': 0.10,
                 'chinese': 0.05, 'price': 0.10},
    '数学推理': {'coding': 0.15, 'reasoning': 0.20, 'math': 0.45,
                 'chinese': 0.10, 'price': 0.10},
    '中文应用': {'coding': 0.15, 'reasoning': 0.20, 'math': 0.10,
                 'chinese': 0.45, 'price': 0.10},
    '成本优先': {'coding': 0.15, 'reasoning': 0.15, 'math': 0.15,
                 'chinese': 0.15, 'price': 0.40},
}

print("\n=== 各场景推荐模型 ===")
for name, weights in scenarios.items():
    model, score = recommend_best(eval_df, weights)
    print(f"  {name}: {model} (综合评分: {score:.2f})")
```
